# -------------------------
# Етап збірки
FROM oven/bun:1-alpine AS builder

WORKDIR /app

RUN apk update && apk add --no-cache dumb-init curl ca-certificates

# Копіюємо package.json та bun.lockb
COPY package.json bun.lockb* ./

# Встановлюємо всі залежності (dev + prod)
RUN bun install --frozen-lockfile

# Копіюємо решту проєкту
COPY . .

# Генеруємо Prisma Client
RUN bunx prisma generate

# Збираємо проєкт
RUN bun run build

# -------------------------
# Фінальний легкий образ для production
FROM oven/bun:1-alpine

WORKDIR /app

RUN apk update && apk add --no-cache dumb-init curl ca-certificates

# Копіюємо зібраний проєкт разом із node_modules з builder stage
COPY --from=builder /app . 

# Тепер додаткового bun install не потрібно, всі production-залежності вже встановлені

ENTRYPOINT ["dumb-init", "--"]
CMD ["bun", "run", "start"]
