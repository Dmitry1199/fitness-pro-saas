# Етап збірки
FROM oven/bun:1-alpine AS builder

WORKDIR /app

RUN apk update && apk add --no-cache dumb-init curl ca-certificates

COPY package.json bun.lockb* ./

# Встановлюємо всі залежності (включно з dev)
RUN bun install --frozen-lockfile

COPY . .

# Генеруємо Prisma Client
RUN bunx prisma generate

# Збираємо проєкт
RUN bun run build

# -------------------------
# Фінальний легкий образ
FROM oven/bun:1-alpine

WORKDIR /app

RUN apk update && apk add --no-cache dumb-init curl ca-certificates

# Копіюємо зібраний проєкт і production-залежності з builder-а
COPY --from=builder /app .

# Встановлюємо лише production-залежності (для мінімального розміру)
RUN bun install --frozen-lockfile --production

ENTRYPOINT ["dumb-init", "--"]

CMD ["bun", "run", "start"]
